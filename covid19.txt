package com.example.covid19;

import android.os.Bundle;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

import com.android.volley.Request;
import com.android.volley.RequestQueue;
import com.android.volley.Response;
import com.android.volley.VolleyError;
import com.android.volley.toolbox.JsonObjectRequest;
import com.android.volley.toolbox.Volley;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;

public class MainActivity extends AppCompatActivity {
TextView Redd,Recov,Death,cases,casesToday,RocrveToDay;
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate ( savedInstanceState );
        setContentView ( R.layout.activity_main );
        Redd = (TextView) findViewById ( R.id.tv_Redd );
        Recov = (TextView) findViewById ( R.id.tvRecov );
        Death = (TextView) findViewById ( R.id.tv_dearh );
        cases = (TextView) findViewById ( R.id.tv_cases );
        casesToday = (TextView) findViewById ( R.id.tv_dayCases );
        RocrveToDay = (TextView) findViewById ( R.id.tv_ReccvrTod );
        RequestQueue requestQueue = Volley.newRequestQueue ( this );
        String url = "https://disease.sh/v3/covid-19/all";
        JsonObjectRequest jsonObjectRequest = new JsonObjectRequest
                (Request.Method.GET, url, null, new Response.Listener<JSONObject>() {

                    @Override
                    public void onResponse(JSONObject response) {
                      String contry="all";
                      try {


                          JSONArray jsonArray = response.getJSONArray ( contry );
                          JSONObject jsonObject = jsonArray.getJSONObject ( jsonArray.length () - 1 );
                          String Redds = jsonObject.getString ( "updated" );
                          String Recovs = jsonObject.getString ( "recovered" );
                          String Deaths = jsonObject.getString ( "deaths" );
                          String casess = jsonObject.getString ( "cases" );
                          String cassessTod = jsonObject.getString ( "todayCases" );
                          String Rocretodays = jsonObject.getString ( "todayRecovered" );
                          Redd.setText ( Redds );
                          Recov.setText ( Recovs );
                          Death.setText ( Deaths );
                          cases.setText ( casess );
                          casesToday.setText ( cassessTod );
                          RocrveToDay.setText ( Rocretodays );


                      }catch (JSONException e)
                      {
                          e.printStackTrace ();
                      }
                    }
                }, new Response.ErrorListener() {

                    @Override
                    public void onErrorResponse(VolleyError error) {
                        // TODO: Handle error

                    }
                });
        requestQueue.add(jsonObjectRequest);

    }

}
